    /**
     * Grava um vetor em um arquivo INI
     */
    function ini_put_contents($inifile, $content)
    {
        $linhas = '';
        foreach ($content as $key => $content)
        {
            $linhas .= "{$key}  =  \"{$content}\"\n";
        }
        file_put_contents($inifile, $linhas);
    }
    ?>
